    function SetupM1Reach()
        local Character = M1ReachPlayer.Character or M1ReachPlayer.CharacterAdded:Wait()
        local RootPart = Character:WaitForChild("HumanoidRootPart")
        
        local function FindClosestTarget()
            if not M1ReachEnabled then
                return nil
            end
            local ClosestTarget = nil
            local ClosestDistance = M1ReachDistance
            for _, OtherPlayer in pairs(game.Players:GetPlayers()) do
                if OtherPlayer ~= M1ReachPlayer and OtherPlayer.Character then
                    local OtherRootPart = OtherPlayer.Character:FindFirstChild("HumanoidRootPart")
                    if OtherRootPart then
                        local Distance = (OtherRootPart.Position - RootPart.Position).magnitude
                        if Distance <= M1ReachDistance and Distance < ClosestDistance then
                            ClosestDistance = Distance
                            ClosestTarget = OtherRootPart
                        end
                    end
                end
            end
            return ClosestTarget
        end
        
        local function TweenToPosition(Position)
            TweenService:Create(RootPart, TweenInfo.new(0.09, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {
                CFrame = Position,
            }):Play()
        end
        
        Character:WaitForChild("Humanoid").AnimationPlayed:Connect(function(AnimationTrack)
            if not M1ReachEnabled then
                return
            end
            if table.find(M1ReachMoves, AnimationTrack.Animation.AnimationId) then
                local Target = FindClosestTarget()
                if Target then
                    local LookVector = Target.CFrame.LookVector
                    local DotProduct = LookVector:Dot((RootPart.Position - Target.Position).unit)
                    local TargetPosition = Target.Position
                    local Offset = nil
                    if DotProduct > 0 then
                        Offset = 0.3
                    else
                        Offset = -0.3
                    end
                    local NewCFrame = CFrame.lookAt(TargetPosition + LookVector * Offset, Target.Position)
                    TweenToPosition(NewCFrame)
                end
            end
        end)
    end

    function CheckCounterMoves()
        local LocalPlayer = CounterNotifierPlayers.LocalPlayer
        local Character = LocalPlayer.Character
        if not Character then
            return
        end
        local RootPart = Character:FindFirstChild("HumanoidRootPart")
        if not RootPart then
            return
        end
        for _, Player in pairs(CounterNotifierPlayers:GetPlayers()) do
            if Player ~= LocalPlayer and Player.Character then
                local OtherCharacter = Player.Character
                local OtherRootPart = OtherCharacter:FindFirstChild("HumanoidRootPart")
                local OtherHumanoid = OtherCharacter:FindFirstChild("Humanoid")
                if OtherRootPart and OtherHumanoid and (RootPart.Position - OtherRootPart.Position).Magnitude <= CounterRange then
                    local CounterAnimation = GetCounterAnimation(OtherHumanoid)
                    if CounterAnimation then
                        local CurrentTime = tick()
                        if not CounterNotifierCooldowns[Player] or 5 < CurrentTime - CounterNotifierCooldowns[Player] then
                            CounterNotifierCooldowns[Player] = CurrentTime
                            local HighlightColor = CounterAnimation == WhiteCounterAnimation and Color3.fromRGB(240, 240, 240) or Color3.fromRGB(0, 0, 254)
                            AddCounterHighlight(OtherCharacter, HighlightColor)
                            SendNotification("Notifier", Player.Name .. " used a counter move", 1.2)
                            task.delay(17.5, function()
                                RemoveCounterHighlight(OtherCharacter)
                                SendNotification("Notifier", Player.Name .. "'s counter move cooldown has ended", 2)
                            end)
                        end
                    end
                end
            end
        end
    end
